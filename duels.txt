local function isMyPlotByName(pn)
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return false end
    local plot = plots:FindFirstChild(pn)
    if not plot then return false end
    local sign = plot:FindFirstChild("PlotSign")
    if sign then
        local yb = sign:FindFirstChild("YourBase")
        if yb and yb:IsA("BillboardGui") then return yb.Enabled end
    end
    return false
end

local function findNearestPrompt()
    local myHrp = getHRP()
    if not myHrp then return nil end
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return nil end
    local np, nd, nn = nil, math.huge, nil
    for _, plot in ipairs(plots:GetChildren()) do
        if isMyPlotByName(plot.Name) then continue end
        local podiums = plot:FindFirstChild("AnimalPodiums")
        if not podiums then continue end
        for _, pod in ipairs(podiums:GetChildren()) do
            pcall(function()
                local base = pod:FindFirstChild("Base")
                local spawn = base and base:FindFirstChild("Spawn")
                if spawn then
                    local dist = (spawn.Position - myHrp.Position).Magnitude
                    if dist < nd and dist <= Values.STEAL_RADIUS then
                        local att = spawn:FindFirstChild("PromptAttachment")
                        if att then
                            for _, ch in ipairs(att:GetChildren()) do
                                if ch:IsA("ProximityPrompt") then
                                    np, nd, nn = ch, dist, pod.Name
                                    break
                                end
                            end
                        end
                    end
                end
            end)
        end
    end
    return np, nd, nn
end

local function ResetProgressBar()
    if Objects.ProgressLabel then Objects.ProgressLabel.Text = "READY" end
    if Objects.ProgressPercentLabel then Objects.ProgressPercentLabel.Text = "" end
    if Objects.ProgressBarFill then Objects.ProgressBarFill.Size = UDim2.new(0, 0, 1, 0) end
end

local function executeSteal(prompt, name)
    if State.isStealing then return end
    if not Refs.StealData[prompt] then
        Refs.StealData[prompt] = {hold = {}, trigger = {}, ready = true}
        pcall(function()
            if getconnections then
                for _, c in ipairs(getconnections(prompt.PromptButtonHoldBegan)) do
                    if c.Function then table.insert(Refs.StealData[prompt].hold, c.Function) end
                end
                for _, c in ipairs(getconnections(prompt.Triggered)) do
                    if c.Function then table.insert(Refs.StealData[prompt].trigger, c.Function) end
                end
            end
        end)
    end
    local data = Refs.StealData[prompt]
    if not data.ready then return end
    data.ready = false
    State.isStealing = true
    State.stealStartTime = tick()
    if Objects.ProgressLabel then Objects.ProgressLabel.Text = name or "STEALING..." end
    if Objects.progressConnection then Objects.progressConnection:Disconnect() end
    Objects.progressConnection = RunService.Heartbeat:Connect(function()
        if not State.isStealing then Objects.progressConnection:Disconnect() return end
        local prog = math.clamp((tick() - State.stealStartTime) / Values.STEAL_DURATION, 0, 1)
        if Objects.ProgressBarFill then Objects.ProgressBarFill.Size = UDim2.new(prog, 0, 1, 0) end
        if Objects.ProgressPercentLabel then Objects.ProgressPercentLabel.Text = math.floor(prog * 100) .. "%" end
    end)
    task.spawn(function()
        for _, f in ipairs(data.hold) do task.spawn(f) end
        local elapsed = 0
        while elapsed < Values.STEAL_DURATION do elapsed = elapsed + task.wait() end
        for _, f in ipairs(data.trigger) do task.spawn(f) end
        if Objects.progressConnection then Objects.progressConnection:Disconnect() end
        ResetProgressBar()
        data.ready = true
        State.isStealing = false
    end)
end

        if not char then return end
        local root = char:FindFirstChild("HumanoidRootPart")

        if hum then
            local st = hum:GetState()
            if st == Enum.HumanoidStateType.Physics or st == Enum.HumanoidStateType.Ragdoll or st == Enum.HumanoidStateType.FallingDown then
                hum:ChangeState(Enum.HumanoidStateType.Running)
                workspace.CurrentCamera.CameraSubject = hum
                pcall(function()
                    local pm = LocalPlayer.PlayerScripts:FindFirstChild("PlayerModule")

            end
        end
        for _, obj in ipairs(char:GetDescendants()) do
            if obj:IsA("Motor6D") and not obj.Enabled then obj.Enabled = true end
        end
    end)
end

local function setupGalaxyForce()
    pcall(function()
        local c = LocalPlayer.Character
        if not c then return end
        local rp = c:FindFirstChild("HumanoidRootPart")
        if not rp then return end
        if Objects.galaxyVectorForce then Objects.galaxyVectorForce:Destroy() end
        if Objects.galaxyAttachment then Objects.galaxyAttachment:Destroy() end
        Objects.galaxyAttachment = Instance.new("Attachment")
        Objects.galaxyAttachment.Parent = rp
        Objects.galaxyVectorForce = Instance.new("VectorForce")
        Objects.galaxyVectorForce.Attachment0 = Objects.galaxyAttachment
        Objects.galaxyVectorForce.ApplyAtCenterOfMass = true
        Objects.galaxyVectorForce.RelativeTo = Enum.ActuatorRelativeTo.World
        Objects.galaxyVectorForce.Force = Vector3.new(0, 0, 0)
        Objects.galaxyVectorForce.Parent = rp
    end)
end

local function updateGalaxyForce()
    if not State.galaxyEnabled or not Objects.galaxyVectorForce then return end
    local c = LocalPlayer.Character
    if not c then return end
    local mass = 0
    for _, p in ipairs(c:GetDescendants()) do
        if p:IsA("BasePart") then mass = mass + p:GetMass() end
    end
    local tg = Values.DEFAULT_GRAVITY * (Values.GalaxyGravityPercent / 100)
    Objects.galaxyVectorForce.Force = Vector3.new(0, mass * (Values.DEFAULT_GRAVITY - tg) * 0.95, 0)
end

local function adjustGalaxyJump()
    pcall(function()
        local c = LocalPlayer.Character
        if not c then return end
        local hum = c:FindFirstChildOfClass("Humanoid")
        if not hum then return end
        if not State.galaxyEnabled then hum.JumpPower = State.originalJumpPower return end
        local ratio = math.sqrt((Values.DEFAULT_GRAVITY * (Values.GalaxyGravityPercent / 100)) / Values.DEFAULT_GRAVITY)
        hum.JumpPower = State.originalJumpPower * ratio
    end)
end

local function doMiniHop()
    if not State.hopsEnabled then return end
    pcall(function()
        local c = LocalPlayer.Character
        if not c then return end
        local rp = c:FindFirstChild("HumanoidRootPart")
        local hum = c:FindFirstChildOfClass("Humanoid")
        if not rp or not hum then return end
        if tick() - State.lastHopTime < Values.HOP_COOLDOWN then return end
        State.lastHopTime = tick()
        if hum.FloorMaterial == Enum.Material.Air then
            rp.AssemblyLinearVelocity = Vector3.new(rp.AssemblyLinearVelocity.X, Values.HOP_POWER, rp.AssemblyLinearVelocity.Z)
        end
    end)
end

local function startFloat()
    State.floatEnabled = true
    setupFloatForce()
    State.cachedMass = 0
    State.lastMassUpdate = 0
    if Objects.floatConn then Objects.floatConn:Disconnect() end
    Objects.floatConn = RunService.Heartbeat:Connect(function()
        if not State.floatEnabled then return end
        pcall(function()
            local c = LocalPlayer.Character
            if not c then return end
            local rp = c:FindFirstChild("HumanoidRootPart")
            if not rp then return end
            local hum = c:FindFirstChildOfClass("Humanoid")
            if not hum then return end
            local now = tick()
            if now - State.lastMassUpdate > 2 or State.cachedMass == 0 then
                State.cachedMass = 0
                for _, p in ipairs(c:GetDescendants()) do
                    if p:IsA("BasePart") then State.cachedMass = State.cachedMass + p:GetMass() end
                end
                State.lastMassUpdate = now
            end
            local rayParams = RaycastParams.new()
            rayParams.FilterDescendantsInstances = {c}
            rayParams.FilterType = Enum.RaycastFilterType.Exclude
            local result = workspace:Raycast(rp.Position, Vector3.new(0, -500, 0), rayParams)
            local groundY = result and result.Position.Y or (rp.Position.Y - State.floatHeight)
            local targetY = groundY + State.floatHeight
            local diff = targetY - rp.Position.Y
            local kP, kD = 500, 80
            local velY = rp.AssemblyLinearVelocity.Y
            if math.abs(diff) < 0.1 and math.abs(velY) < 0.5 then diff, velY = 0, 0 end
            if Objects.floatForce then
                Objects.floatForce.Force = Vector3.new(0, State.cachedMass * (kP * diff - kD * velY + workspace.Gravity), 0)
            end
            hum.JumpPower = 0
        end)
    end)
end

local function stopFloat()
    State.floatEnabled = false
    State.cachedMass = 0
    if Objects.floatConn then Objects.floatConn:Disconnect() Objects.floatConn = nil end
    if Objects.floatForce then Objects.floatForce:Destroy() Objects.floatForce = nil end
    if Objects.floatAttachment then Objects.floatAttachment:Destroy() Objects.floatAttachment = nil end
    pcall(function()
        local c = LocalPlayer.Character
        if not c then return end
        local hum = c:FindFirstChildOfClass("Humanoid")
        if not hum then return end
        local rp = c:FindFirstChild("HumanoidRootPart")
        if not rp then return end
        hum.JumpPower = State.originalJumpPower
        local rayParams = RaycastParams.new()
        rayParams.FilterDescendantsInstances = {c}
        rayParams.FilterType = Enum.RaycastFilterType.Exclude
        local result = workspace:Raycast(rp.Position, Vector3.new(0, -500, 0), rayParams)
        if result then
            rp.CFrame = CFrame.new(rp.Position.X, result.Position.Y + rp.Size.Y / 2 + 0.1, rp.Position.Z)
            rp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        end
    end)
end

local function isXrayTarget(obj)
    if not obj:IsA("BasePart") then return false end
    if not obj.Anchored then return false end
    local nameLower = obj.Name:lower()
    local parentNameLower = (obj.Parent and obj.Parent.Name:lower()) or ""
    return nameLower:find("base") ~= nil or parentNameLower:find("base") ~= nil
end

local function applyXrayToPart(obj)
    pcall(function()
        if isXrayTarget(obj) and Refs.originalTransparency[obj] == nil then
            Refs.originalTransparency[obj] = obj.LocalTransparencyModifier
            obj.LocalTransparencyModifier = 0.85
        end
    end)
end

local function applyXrayToAll()
    pcall(function()
        for _, obj in ipairs(workspace:GetDescendants()) do applyXrayToPart(obj) end
    end)
end

local function startXrayWatcher()
    if Objects.xrayDescendantConn then return end
    Objects.xrayDescendantConn = workspace.DescendantAdded:Connect(function(obj)
        if not State.xrayEnabled then return end
        task.defer(function() if State.xrayEnabled then applyXrayToPart(obj) end end)
    end)
end

local function stopXrayWatcher()
    if Objects.xrayDescendantConn then Objects.xrayDescendantConn:Disconnect() Objects.xrayDescendantConn = nil end
end

local function enableOptimizer()
    if getgenv and getgenv().OPTIMIZER_ACTIVE then return end
    if getgenv then getgenv().OPTIMIZER_ACTIVE = true end
    pcall(function() setfpscap(999) end)
    pcall(function()
        Lighting.GlobalShadows = false
        Lighting.Brightness = 3
        Lighting.FogEnd = 9e9
    end)
    pcall(function()
        for _, obj in ipairs(workspace:GetDescendants()) do
            pcall(function()
                if obj:IsA("ParticleEmitter") or obj:IsA("Trail") or obj:IsA("Beam") then
                    obj:Destroy()
                elseif obj:IsA("BasePart") then
                    obj.CastShadow = false
                    obj.Material = Enum.Material.Plastic
                end
            end)
        end
    end)
    State.xrayEnabled = true
    applyXrayToAll()
    startXrayWatcher()
end

local function disableOptimizer()
    if getgenv then getgenv().OPTIMIZER_ACTIVE = false end
    stopXrayWatcher()
    if State.xrayEnabled then
        for part, val in pairs(Refs.originalTransparency) do
            if part and part.Parent then part.LocalTransparencyModifier = val end
        end
        Refs.originalTransparency = {}
        State.xrayEnabled = false
    end
end

local function startAutoLeft()
    if Objects.autoLeftConnection then Objects.autoLeftConnection:Disconnect() end
    State.autoLeftPhase = 1
    Objects.autoLeftConnection = RunService.Heartbeat:Connect(function()
        if not State.AutoLeftEnabled then return end
        local c = LocalPlayer.Character
        if not c then return end
        local rp = c:FindFirstChild("HumanoidRootPart")
        local hum = c:FindFirstChildOfClass("Humanoid")
        if not rp or not hum then return end
        local spd = Config.NORMAL_SPEED
        if State.autoLeftPhase == 1 then
            local tgt = Vector3.new(Positions.L1.X, rp.Position.Y, Positions.L1.Z)
            if (tgt - rp.Position).Magnitude < 1 then
                State.autoLeftPhase = 2
                local d = (Positions.L2 - rp.Position)
                local mv = Vector3.new(d.X, 0, d.Z).Unit
                hum:Move(mv, false)
                rp.AssemblyLinearVelocity = Vector3.new(mv.X * spd, rp.AssemblyLinearVelocity.Y, mv.Z * spd)
                return
            end
            local d = (Positions.L1 - rp.Position)
            local mv = Vector3.new(d.X, 0, d.Z).Unit
            hum:Move(mv, false)
            rp.AssemblyLinearVelocity = Vector3.new(mv.X * spd, rp.AssemblyLinearVelocity.Y, mv.Z * spd)
        elseif State.autoLeftPhase == 2 then
            local tgt = Vector3.new(Positions.L2.X, rp.Position.Y, Positions.L2.Z)
            if (tgt - rp.Position).Magnitude < 1 then
                hum:Move(Vector3.zero, false)
                rp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
                State.AutoLeftEnabled = false
                Enabled.AutoLeftEnabled = false
                if Objects.autoLeftConnection then Objects.autoLeftConnection:Disconnect() Objects.autoLeftConnection = nil end
                State.autoLeftPhase = 1
                local sw = Refs.Switches.AutoLeft
                if sw then
                    sw.bg.BackgroundColor3 = Colors.SW_OFF
                    sw.circle.Position = UDim2.new(0, 2, 0.5, -7)
                end
                updateSideButtons()
                faceSouth()
                return
            end
            local d = (Positions.L2 - rp.Position)
            local mv = Vector3.new(d.X, 0, d.Z).Unit
            hum:Move(mv, false)
            rp.AssemblyLinearVelocity = Vector3.new(mv.X * spd, rp.AssemblyLinearVelocity.Y, mv.Z * spd)
        end
    end)
end

local function stopAutoLeft()

    local c = LocalPlayer.Character
    if c then
        local hum = c:FindFirstChildOfClass("Humanoid")

    end
end

local function getBat()
    local char = LocalPlayer.Character
    if not char then return nil end
    local tool = char:FindFirstChild("Bat")
    if tool then return tool end
    local backpack = LocalPlayer:FindFirstChild("Backpack")
    if backpack then
        tool = backpack:FindFirstChild("Bat")
        if tool then tool.Parent = char return tool end
    end
    return nil
end

local function tryHitBat()
    if State.hittingCooldown then return end
    State.hittingCooldown = true
    local bat = getBat()
    if bat then
        pcall(function()
            bat:Activate()
            local evt = bat:FindFirstChildWhichIsA("RemoteEvent")
            if evt then evt:FireServer() end
        end)
    end
    task.delay(0.08, function() State.hittingCooldown = false end)
end

local function getClosestPlayer()
    local closestPlayer, closestDist = nil, math.huge
    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            local targetHRP = plr.Character.HumanoidRootPart
            local dist = (Objects.hrp.Position - targetHRP.Position).Magnitude
            if dist < closestDist then closestDist = dist closestPlayer = plr end
        end
    end
    return closestPlayer, closestDist
end

local function flyToFrontOfTarget(targetHRP)
    if not Objects.hrp then return end
    local forward = targetHRP.CFrame.LookVector
    local frontPos = targetHRP.Position + forward * 1.5
    local direction = (frontPos - Objects.hrp.Position).Unit
    Objects.hrp.Velocity = Vector3.new(direction.X * 56.5, direction.Y * 56.5, direction.Z * 56.5)
end

        floatHeight = State.floatHeight,
        speedToggled = State.speedToggled,
        autoBatToggled = State.autoBatToggled,
    }
    local ok = false
    if writefile then
        local success = pcall(function()
            writefile("SpearDuelsConfig.json", HttpService:JSONEncode(cfg))
        end)
        ok = success
    end
    if Objects.saveConfigBtn then
        local prev = Objects.saveConfigBtn.Text
        Objects.saveConfigBtn.Text = ok and "SAVED" or "FAILED"
        task.wait(1.5)
        if Objects.saveConfigBtn and Objects.saveConfigBtn.Parent then
            Objects.saveConfigBtn.Text = prev
        end
    end
    return ok
end

local function loadConfig()
    if not (isfile and isfile("SpearDuelsConfig.json")) then return false end
    local ok, cfg = pcall(function()
        return HttpService:JSONDecode(readfile("SpearDuelsConfig.json"))
    end)
    if not ok or not cfg then return false end
    if cfg.normalSpeed and type(cfg.normalSpeed) == "number" then Config.NORMAL_SPEED = cfg.normalSpeed end
    if cfg.carrySpeed and type(cfg.carrySpeed) == "number" then Config.CARRY_SPEED = cfg.carrySpeed end
    if cfg.fovValue and type(cfg.fovValue) == "number" then Config.FOV_VALUE = cfg.fovValue end
    if cfg.uiScale and type(cfg.uiScale) == "number" then Config.UI_SCALE = cfg.uiScale end
    if cfg.autoBatKey and Enum.KeyCode[cfg.autoBatKey] then Keybinds.AutoBat = Enum.KeyCode[cfg.autoBatKey] end
    if cfg.speedToggleKey and Enum.KeyCode[cfg.speedToggleKey] then Keybinds.SpeedToggle = Enum.KeyCode[cfg.speedToggleKey] end
    if cfg.autoLeftKey and Enum.KeyCode[cfg.autoLeftKey] then Keybinds.AutoLeft = Enum.KeyCode[cfg.autoLeftKey] end
    if cfg.autoRightKey and Enum.KeyCode[cfg.autoRightKey] then Keybinds.AutoRight = Enum.KeyCode[cfg.autoRightKey] end
    if cfg.infiniteJumpKey and Enum.KeyCode[cfg.infiniteJumpKey] then Keybinds.InfiniteJump = Enum.KeyCode[cfg.infiniteJumpKey] end
    if cfg.floatKey and Enum.KeyCode[cfg.floatKey] then Keybinds.Float = Enum.KeyCode[cfg.floatKey] end
    if cfg.grabRadius and type(cfg.grabRadius) == "number" then Values.STEAL_RADIUS = cfg.grabRadius end
    if cfg.stealDuration and type(cfg.stealDuration) == "number" then Values.STEAL_DURATION = cfg.stealDuration end
    if cfg.galaxyGravity and type(cfg.galaxyGravity) == "number" then Values.GalaxyGravityPercent = cfg.galaxyGravity end
    if cfg.hopPower and type(cfg.hopPower) == "number" then Values.HOP_POWER = cfg.hopPower end
    if cfg.antiRagdoll ~= nil then Enabled.AntiRagdoll = cfg.antiRagdoll end
    if cfg.autoStealEnabled ~= nil then Enabled.AutoSteal = cfg.autoStealEnabled end
    if cfg.infiniteJump ~= nil then Enabled.InfiniteJump = cfg.infiniteJump end
    if cfg.optimizer ~= nil then Enabled.Optimizer = cfg.optimizer end
    if cfg.unwalk ~= nil then Enabled.Unwalk = cfg.unwalk end
    if cfg.shinyGraphics ~= nil then Enabled.ShinyGraphics = cfg.shinyGraphics end
    if cfg.autoLeftEnabled ~= nil then Enabled.AutoLeftEnabled = cfg.autoLeftEnabled end
    if cfg.autoRightEnabled ~= nil then Enabled.AutoRightEnabled = cfg.autoRightEnabled end
    if cfg.floatEnabled ~= nil then State.floatEnabled = cfg.floatEnabled end
    if cfg.floatHeight and type(cfg.floatHeight) == "number" then State.floatHeight = cfg.floatHeight end
    if cfg.speedToggled ~= nil then State.speedToggled = cfg.speedToggled end
    if cfg.autoBatToggled ~= nil then State.autoBatToggled = cfg.autoBatToggled end
    return true
end

local function makeSlider(parent, lbl, minVal, maxVal, defaultVal, onChanged)
    local wrap = Instance.new("Frame", parent)
    wrap.Size = UDim2.new(1, 0, 0, Config.isMobile and 34 or 38)
    wrap.BackgroundColor3 = Colors.ROW
    wrap.BorderSizePixel = 0
    wrap.ZIndex = 3
    Instance.new("UICorner", wrap).CornerRadius = UDim.new(0, 8)
    local lbTxt = Instance.new("TextLabel", wrap)
    lbTxt.Size = UDim2.new(0.28, 0, 1, 0)
    lbTxt.Position = UDim2.new(0, 12, 0, 0)
    lbTxt.BackgroundTransparency = 1
    lbTxt.Text = lbl
    lbTxt.TextColor3 = Colors.TEXT_DIM
    lbTxt.Font = Enum.Font.GothamMedium
    lbTxt.TextSize = Config.isMobile and 11 or 12
    lbTxt.TextXAlignment = Enum.TextXAlignment.Left
    lbTxt.ZIndex = 4
    local valTxt = Instance.new("TextLabel", wrap)
    valTxt.Size = UDim2.new(0, 40, 0, 18)
    valTxt.Position = UDim2.new(1, -50, 0.5, -9)
    valTxt.BackgroundTransparency = 1
    valTxt.Text = tostring(math.floor(defaultVal))
    valTxt.TextColor3 = Colors.WHITE
    valTxt.Font = Enum.Font.GothamMedium
    valTxt.TextSize = Config.isMobile and 11 or 12
    valTxt.TextXAlignment = Enum.TextXAlignment.Right
    valTxt.ZIndex = 5
    local trackWidth = Config.isMobile and 120 or 160
    local trackBg = Instance.new("Frame", wrap)
    trackBg.Size = UDim2.new(0, trackWidth, 0, 4)
    trackBg.Position = UDim2.new(0.32, 0, 0.5, -2)
    trackBg.BackgroundColor3 = Colors.SW_OFF
    trackBg.BorderSizePixel = 0
    trackBg.ZIndex = 4
    Instance.new("UICorner", trackBg).CornerRadius = UDim.new(1, 0)
    local trackFill = Instance.new("Frame", trackBg)
    trackFill.Size = UDim2.new((defaultVal - minVal) / (maxVal - minVal), 0, 1, 0)
    trackFill.BackgroundColor3 = Colors.ACCENT2
    trackFill.BorderSizePixel = 0
    trackFill.ZIndex = 5
    Instance.new("UICorner", trackFill).CornerRadius = UDim.new(1, 0)
    local thumb = Instance.new("TextButton", trackBg)
    thumb.Size = UDim2.new(0, Config.isMobile and 14 or 12, 0, Config.isMobile and 14 or 12)
    thumb.Position = UDim2.new((defaultVal - minVal) / (maxVal - minVal), Config.isMobile and -7 or -6, 0.5, Config.isMobile and -7 or -6)
    thumb.BackgroundColor3 = Colors.WHITE
    thumb.BorderSizePixel = 0
    thumb.ZIndex = 6
    thumb.Text = ""
    thumb.AutoButtonColor = false
    Instance.new("UICorner", thumb).CornerRadius = UDim.new(1, 0)
    local dragging, dragConnection, lastUpdate = false, nil, 0
    local function updateSlider(inputPos)
        local relX = math.clamp((inputPos.X - trackBg.AbsolutePosition.X) / trackBg.AbsoluteSize.X, 0, 1)
        local value = minVal + (maxVal - minVal) * relX
        trackFill.Size = UDim2.new(relX, 0, 1, 0)
        thumb.Position = UDim2.new(relX, Config.isMobile and -7 or -6, 0.5, Config.isMobile and -7 or -6)
        if lbl == "FOV" then valTxt.Text = tostring(math.floor(value))
        elseif lbl == "UI Scale" then valTxt.Text = string.format("%.1f", value)
        else valTxt.Text = tostring(math.floor(value)) end
        if lbl == "UI Scale" then
            local now = tick()
            if now - lastUpdate >= 0.08 or not dragging then lastUpdate = now if onChanged then onChanged(value) end end
        else if onChanged then onChanged(value) end end
    end
    thumb.MouseButton1Down:Connect(function()
        dragging = true
        if dragConnection then dragConnection:Disconnect() end
        dragConnection = UserInputService.InputChanged:Connect(function(inp)
            if inp.UserInputType == Enum.UserInputType.MouseMovement then updateSlider(inp.Position) end
        end)
    end)
    thumb.InputBegan:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            if dragConnection then dragConnection:Disconnect() end
            dragConnection = inp.Changed:Connect(function()
                if inp.UserInputState == Enum.UserInputState.End then
                    dragging = false if dragConnection then dragConnection:Disconnect() dragConnection = nil end return
                end
                updateSlider(inp.Position)
            end)
        end
    end)
    trackBg.InputBegan:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.Touch or inp.UserInputType == Enum.UserInputType.MouseButton1 then updateSlider(inp.Position) end
    end)
    UserInputService.InputEnded:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false if dragConnection then dragConnection:Disconnect() dragConnection = nil end
        end
    end)
    return wrap
end

-- Input Handler
UserInputService.InputBegan:Connect(function(input, gpe)
    if gpe then return end
    if State.waitingForKey then return end
    if input.KeyCode == Keybinds.UIToggle then
        Objects.main.Visible = not Objects.main.Visible
        TweenService:Create(Objects.mobileToggleBtn, TweenInfo.new(0.2), {BackgroundColor3 = Objects.main.Visible and Colors.ROW_HOVER or Colors.CARD}):Play()
    end
    if input.KeyCode == Keybinds.SpeedToggle then
        State.speedToggled = not State.speedToggled
        local sw = Refs.Switches.Speed
        if sw then
            TweenService:Create(sw.bg, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {BackgroundColor3 = State.speedToggled and Colors.SW_ON or Colors.SW_OFF}):Play()
            TweenService:Create(sw.circle, TweenInfo.new(0.2, Enum.EasingStyle.Back), {Position = State.speedToggled and UDim2.new(1, -16, 0.5, -7) or UDim2.new(0, 2, 0.5, -7)}):Play()
            TweenService:Create(sw.circle, TweenInfo.new(0.2), {BackgroundColor3 = State.speedToggled and Colors.BG or Colors.ACCENT2}):Play()
        end
        updateSideButtons()
    end
    if input.KeyCode == Keybinds.AutoBat then
        State.autoBatToggled = not State.autoBatToggled
        local sw = Refs.Switches.Bat
        if sw then
            TweenService:Create(sw.bg, TweenInfo.new(0.2, Enum.EasingStyle.Quint), {BackgroundColor3 = State.autoBatToggled and Colors.SW_ON or Colors.SW_OFF}):Play()
            TweenService:Create(sw.circle, TweenInfo.new(0.2, Enum.EasingStyle.Back), {Position = State.autoBatToggled and UDim2.new(1, -16, 0.5, -7) or UDim2.new(0, 2, 0.5, -7)}):Play()
            TweenService:Create(sw.circle, TweenInfo.new(0.2), {BackgroundColor3 = State.autoBatToggled and Colors.BG or Colors.ACCENT2}):Play()
        end
    end
    if input.KeyCode == Keybinds.AutoLeft then
        State.AutoLeftEnabled = not State.AutoLeftEnabled
        Enabled.AutoLeftEnabled = State.AutoLeftEnabled
        if Refs.VisualSetters.AutoLeftEnabled then Refs.VisualSetters.AutoLeftEnabled(State.AutoLeftEnabled) end
        if State.AutoLeftEnabled then startAutoLeft() else stopAutoLeft() end
        updateSideButtons()
    end
    if input.KeyCode == Keybinds.AutoRight then
        State.AutoRightEnabled = not State.AutoRightEnabled
        Enabled.AutoRightEnabled = State.AutoRightEnabled
        if Refs.VisualSetters.AutoRightEnabled then Refs.VisualSetters.AutoRightEnabled(State.AutoRightEnabled) end
        if State.AutoRightEnabled then startAutoRight() else stopAutoRight() end
        updateSideButtons()
    end
    if input.KeyCode == Keybinds.InfiniteJump then
        Enabled.InfiniteJump = not Enabled.InfiniteJump
        if Refs.VisualSetters.InfiniteJump then Refs.VisualSetters.InfiniteJump(Enabled.InfiniteJump) end
        if Enabled.InfiniteJump then startInfiniteJump() else stopInfiniteJump() end
    end
    if input.KeyCode == Keybinds.Float then
        State.floatEnabled = not State.floatEnabled
        if Objects.floatVisualSetter then Objects.floatVisualSetter(State.floatEnabled) end
        if State.floatEnabled then startFloat() else stopFloat() end
    end
    if input.KeyCode == Enum.KeyCode.Space then State.spaceHeld = true end
end)

UserInputService.InputEnded:Connect(function(input)
    if input.KeyCode == Enum.KeyCode.Space then State.spaceHeld = false end
end)

-- Movement Loop
RunService.RenderStepped:Connect(function()
    if not (Objects.h and Objects.hrp) then return end
    if not (State.AutoLeftEnabled or State.AutoRightEnabled) then
        local md = Objects.h.MoveDirection
        local spd = State.speedToggled and Config.CARRY_SPEED or Config.NORMAL_SPEED
        if md.Magnitude > 0 then Objects.hrp.Velocity = Vector3.new(md.X * spd, Objects.hrp.Velocity.Y, md.Z * spd) end
    end
    if Objects.speedLbl then Objects.speedLbl.Text = "Speed: " .. string.format("%.1f", Vector3.new(Objects.hrp.Velocity.X, 0, Objects.hrp.Velocity.Z).Magnitude) end
end)

RunService.Heartbeat:Connect(function()
    if State.autoBatToggled and Objects.h and Objects.hrp then
        local target, dist = getClosestPlayer()
        if target and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
            flyToFrontOfTarget(target.Character.HumanoidRootPart)
            if dist <= 5 then tryHitBat() end
        end
    end
end)

-- Apply Loaded Settings
task.spawn(function()
    task.wait(2)
    if Enabled.AutoSteal and Refs.VisualSetters.AutoSteal then Refs.VisualSetters.AutoSteal(true) startAutoSteal() end
    if Enabled.AntiRagdoll and Refs.VisualSetters.AntiRagdoll then Refs.VisualSetters.AntiRagdoll(true) startAntiRagdoll() end
    if Enabled.InfiniteJump and Refs.VisualSetters.InfiniteJump then Refs.VisualSetters.InfiniteJump(true) startInfiniteJump() end
    if Enabled.Unwalk and Refs.VisualSetters.Unwalk then Refs.VisualSetters.Unwalk(true) startUnwalk() end
    if Enabled.Optimizer and Refs.VisualSetters.Optimizer then Refs.VisualSetters.Optimizer(true) enableOptimizer() end
    if Enabled.ShinyGraphics and Refs.VisualSetters.ShinyGraphics then Refs.VisualSetters.ShinyGraphics(true) enableShinyGraphics() end
    if Enabled.AutoLeftEnabled then
        State.AutoLeftEnabled = true
        if Refs.VisualSetters.AutoLeftEnabled then Refs.VisualSetters.AutoLeftEnabled(true) end
        startAutoLeft()
    end
    if Enabled.AutoRightEnabled then
        State.AutoRightEnabled = true
        if Refs.VisualSetters.AutoRightEnabled then Refs.VisualSetters.AutoRightEnabled(true) end
        startAutoRight()
    end
    if State.floatEnabled then
        if Objects.floatVisualSetter then Objects.floatVisualSetter(true) end
        startFloat()
    end
    if State.speedToggled then
        local sw = Refs.Switches.Speed
        if sw then
            sw.bg.BackgroundColor3 = Colors.SW_ON
            sw.circle.Position = UDim2.new(1, -16, 0.5, -7)
            sw.circle.BackgroundColor3 = Colors.BG
        end
    end
    if State.autoBatToggled then
        local sw = Refs.Switches.Bat
        if sw then
            sw.bg.BackgroundColor3 = Colors.SW_ON
            sw.circle.Position = UDim2.new(1, -16, 0.5, -7)
            sw.circle.BackgroundColor3 = Colors.BG
        end
    end
    updateSideButtons()
    applyFOV(Config.FOV_VALUE)
    applyUIScale(Config.UI_SCALE)
end)

local function setupChar(char)
    Objects.h = char:WaitForChild("Humanoid")
    Objects.hrp = char:WaitForChild("HumanoidRootPart")
    local head = char:FindFirstChild("Head")
    if head then
        local bb = Instance.new("BillboardGui", head)
        bb.Size = UDim2.new(0, 120, 0, 22)
        bb.StudsOffset = Vector3.new(0, 2.5, 0)
        bb.AlwaysOnTop = true
        Objects.speedLbl = Instance.new("TextLabel", bb)
        Objects.speedLbl.Size = UDim2.new(1, 0, 1, 0)
        Objects.speedLbl.BackgroundTransparency = 1
        Objects.speedLbl.TextColor3 = Colors.WHITE
        Objects.speedLbl.Font = Enum.Font.GothamBold
        Objects.speedLbl.TextScaled = true
        Objects.speedLbl.TextStrokeTransparency = 0.3
        Objects.speedLbl.TextStrokeColor3 = Colors.BG
    end
    if State.floatEnabled then setupFloatForce() end
end
